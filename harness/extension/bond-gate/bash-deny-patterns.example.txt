# One pattern per line. Blank lines and # comments are ignored.
# Patterns are treated as regex first; if a line is not valid regex,
# bash-policy falls back to simple case-insensitive substring match.

# Block daemon/control HTTP bypasses routed through other hosts or ports
curl\s+.*(/api/(health|questions|missions|emissions|sessions|bonds|entities|channels))
wget\s+.*(/api/(health|questions|missions|emissions|sessions|bonds|entities|channels))

# Block destructive host operations
rm\s+-rf\s+/
shutdown\b
reboot\b
systemctl\s+

# Block publish/release flows that should route through kingdom tools or Vulcan
npm\s+publish
meteor\s+publish
cargo\s+publish

# Example literal fallback — if you don't need regex, plain text works too
secret.txt
